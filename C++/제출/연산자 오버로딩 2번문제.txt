class Time
{
private:
	int m_iHour;
	int m_iMin;
public:
	inline void ShowTime()	{cout << " 총 공부 시간 : " << m_iHour << " : " << m_iMin << endl;}
	void operator +(Time time);
	Time()
	{
		m_iHour = 0;
		m_iMin = 0;
	}
	Time(int Hour, int Min)
	{
		m_iHour = Hour;
		m_iMin = Min;
	}
};

void Time ::operator +(Time time)
{
	m_iHour += time.m_iHour;
	m_iMin += time.m_iMin;
	while(m_iMin >= 60)
	{
		m_iMin -= 60;
		m_iHour++;
	}
}
