#include <stdio.h>
#include <string.h>

 

class CString
{
    char *data;
    int      len;

 

public:
    CString( const char* string = "" )
    {
        len = strlen( string );
        data = new char[len+1];
    }

 

    ~CString()
    {
        delete [] data;
    }
};

 

void main( void )
{
    CString string1( "hello" );      // 생성자가 호출
    CString string2 = string1;       // 생성자가 호출되지 않음
}
