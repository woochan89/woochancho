#include<iostream>
#include<stdio.h>
using namespace std;
int atk = 100;
int cp = 0;
float cd = 1.25;

int tmpatk = 100;
int tmpcp = 0;
float tmpcd = 1.25;
float tmpsum = 100000;

void cal() {
	int num;
	float tmp;
	float sum=0;
	for (int i = 0; i < 1000; i++) {
		num = rand();
		if (num % 100 < cp)
			tmp = (float)atk * cd;
		else
			tmp = atk;
		sum += tmp;
	}
	printf("공 : %d\n회심률 : %d\n슈회 : %0.2f\n1000회 평균값 : %0.2f\n\n",atk,cp,cd,sum / 1000);
	printf("공 : %d\n회심률 : %d\n슈회 : %0.2f\n1000회 평균값 : %0.2f\n\n", tmpatk, tmpcp, tmpcd, tmpsum / 1000);

	printf("현재 값을 저장하시겠습니까?(1:yes  0:no)\n");
	scanf("%d", &num);
	if (num == 1)
	{
		tmpatk = atk;
		tmpcp = cp;
		tmpcd = cd;
		tmpsum = sum;
	}

}

int main()
{
	int num;
	while (1) {
		system("cls");
		printf("1.공격력\n");
		printf("2.회심률\n");
		printf("3.슈회\n");
		printf("4.계산\n");
		printf("5.종료\n");
		scanf("%d", &num);
		switch (num) {
		case 1:
			printf("기존 공격력 %d\n", atk);
			scanf("%d", &atk);
			break;
		case 2:
			printf("기존 회심률 %d\n", cp);
			scanf("%d", &cp);
			break;
		case 3:
			printf("기존 슈회 %0.2f\n", cd);
			scanf("%f", &cd);
			break;
		case 4:
			cal();
			break;
		case 5:
			return 0;
		}
	}
	return 0;
}
