#include <iostream>
using namespace std;

int Fibo(int n)
{
	if (n == 1)
		return 0;
	else if (n == 2)
		return 1;
	else
		return Fibo(n - 1) + Fibo(n - 2);
}

int main()
{
	int Num;
	while (true)
	{
		system("cls");
		cout << "원하는 수를 입력해주세요 : ";
		cin >> Num;
		if (Num < 0)
		{
			cout << "0이상의 숫자를 입력해주세요"<<endl;
			system("pause");
			continue;
		}
		cout << "F" << Num << " = " << Fibo(Num)<<endl;
		system("pause");
	}
}
